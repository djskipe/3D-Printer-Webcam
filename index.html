const CANALE_UNICO = "STAMPANTE_3D_DI_PRO_XYZ"; const config = { 'iceServers': [{ 'urls': 'stun:stun.l.google.com:19302' }] }; const v = document.getElementById('v'); const status = document.getElementById('status'); const ui = document.getElementById('ui'); async function startStreamer() { ui.style.display = 'none'; try { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: { ideal: "environment" } }, audio: false }); v.srcObject = stream; const peer = new Peer(CANALE_UNICO, { config: config }); peer.on('open', () => status.innerText = "🔴 TRASMETTENDO..."); peer.on('call', (call) => { status.innerText = "PC COLLEGATO! Invio in corso..."; call.answer(stream); }); } catch (e) { alert("Errore camera: " + e.message); } } function startReceiver() { ui.style.display = 'none'; status.innerText = "In attesa del segnale..."; const peer = new Peer({ config: config }); peer.on('open', () => { // Funzione di chiamata ricorsiva const connect = () => { if (v.srcObject) return; // Se abbiamo già il video, fermati const call = peer.call(CANALE_UNICO, new MediaStream()); call.on('stream', (remoteStream) => { console.log("Stream ricevuto!"); status.innerText = "🔵 VIDEO AGGANCIATO"; v.srcObject = remoteStream; // TRUCCO: Forza l'esecuzione del video v.load(); const playPromise = v.play(); if (playPromise !== undefined) { playPromise.catch(() => { status.innerText = "Clicca ovunque per vedere il video"; window.addEventListener('click', () => v.play(), {once: true}); }); } }); }; // Prova a connettersi ogni 2 secondi const interval = setInterval(() => { if (v.srcObject) clearInterval(interval); else connect(); }, 2000); }); }
